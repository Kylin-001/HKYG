## 2. 安全中间件

### 2.1 安全头部配置

位置: `src/plugins/security.ts`

```typescript
export const securityHeaders = {
  'Content-Security-Policy': contentSecurityPolicy,
  'X-Content-Type-Options': 'nosniff',
  'X-Frame-Options': 'DENY',
  'X-XSS-Protection': '1; mode=block',
  'Referrer-Policy': 'strict-origin-when-cross-origin',
  'Permissions-Policy': 'camera=(), microphone=(), geolocation=()',
  'Strict-Transport-Security': 'max-age=31536000; includeSubDomains; preload',
}
```

### 2.2 XSS防护

```typescript
import { xssProtection } from '@/plugins/security'

// 清理HTML
const safeHtml = xssProtection.sanitizeHtml(userInput)

// 清理URL
const safeUrl = xssProtection.sanitizeUrl(userInput)

// 清理用户输入
const safeInput = xssProtection.sanitizeInput(userInput)
```

### 2.3 CSRF防护

```typescript
import { csrfProtection } from '@/plugins/security'

// 生成CSRF令牌
const token = csrfProtection.generateToken()

// 验证令牌
const isValid = csrfProtection.validateToken(token, sessionToken)
```

## 3. API速率限制

### 3.1 配置

位置: `src/utils/rate-limiter.ts`

```typescript
export const RATE_LIMIT_CONFIGS = {
  // 通用API限制：每分钟60次
  general: { windowMs: 60000, maxRequests: 60 },
  
  // 登录API限制：每分钟5次
  login: { windowMs: 60000, maxRequests: 5 },
  
  // 注册API限制：每小时3次
  register: { windowMs: 3600000, maxRequests: 3 },
  
  // 敏感操作限制：每分钟10次
  sensitive: { windowMs: 60000, maxRequests: 10 },
  
  // 文件上传限制：每分钟5次
  upload: { windowMs: 60000, maxRequests: 5 },
  
  // 搜索限制：每分钟30次
  search: { windowMs: 60000, maxRequests: 30 },
}
```

### 3.2 使用方式

```typescript
import { rateLimiters } from '@/utils/rate-limiter'

// 检查是否允许请求
const isAllowed = rateLimiters.login.isAllowed('user_id_123')

if (!isAllowed) {
  // 提示用户稍后再试
}
```

### 3.3 环境变量配置

```bash
# 速率限制配置
VITE_APP_RATE_LIMIT_WINDOW=60
VITE_APP_RATE_LIMIT_MAX_REQUESTS=60
VITE_APP_LOGIN_RATE_LIMIT_WINDOW=60
VITE_APP_LOGIN_RATE_LIMIT_MAX_REQUESTS=5
```

## 4. 数据脱敏

### 4.1 脱敏工具

位置: `src/utils/data-mask.ts`

```typescript
import { maskUtils } from '@/utils/data-mask'

// 手机号脱敏: 13812345678 -> 138****5678
maskUtils.maskPhone('13812345678')

// 身份证脱敏: 110101199001011234 -> 110***********1234
maskUtils.maskIdCard('110101199001011234')

// 银行卡脱敏: 6222021234567890123 -> 6222 **** **** 0123
maskUtils.maskBankCard('6222021234567890123')

// 邮箱脱敏: example@domain.com -> exa***@domain.com
maskUtils.maskEmail('example@domain.com')

// 姓名脱敏: 张三丰 -> 张*丰
maskUtils.maskName('张三丰')
```

### 4.2 批量脱敏

```typescript
const userData = [
  { name: '张三', phone: '13812345678', idCard: '110101199001011234' },
  { name: '李四', phone: '13998765432', idCard: '310101198505051234' },
]

const maskedData = maskUtils.maskArray(userData, [
  { key: 'name', type: 'name' },
  { key: 'phone', type: 'phone' },
  { key: 'idCard', type: 'idCard' },
])
```

## 5. 安全日志审计

### 5.1 日志记录

位置: `src/utils/security-audit.ts`

```typescript
import securityLogger from '@/utils/security-audit'

// 记录登录
securityLogger.logLogin({
  username: 'user123',
  result: 'success',
})

// 记录敏感操作
securityLogger.logSensitiveOperation({
  userId: '123',
  username: 'user123',
  action: '修改密码',
  result: 'success',
})

// 记录权限拒绝
securityLogger.logPermissionDenied({
  userId: '123',
  username: 'user123',
  action: '访问管理后台',
  resource: '/admin/settings',
})

// 记录速率限制
securityLogger.logRateLimitExceeded({
  ip: '192.168.1.1',
  endpoint: '/api/user/login',
  limit: 5,
  window: '60秒',
})
```

### 5.2 查询日志

```typescript
// 获取最近100条日志
const recentLogs = securityLogger.getRecentLogs(100)

// 获取高风险日志
const highRiskLogs = securityLogger.getHighRiskLogs(50)

// 获取指定用户的日志
const userLogs = securityLogger.getLogsByUser('123')
```
