## 测试命名规范

### 测试套件命名
```typescript
// ✅ 推荐
describe('ComponentName', () => {})
describe('ModuleName.functionName', () => {})

// ❌ 不推荐
describe('test', () => {})
describe('Test Component', () => {})
```

### 测试用例命名
```typescript
// ✅ 推荐
it('should render correctly', () => {})
it('should emit click event when button clicked', () => {})
it('should return empty array when no data', () => {})

// ❌ 不推荐
it('test render', () => {})
it('click test', () => {})
it('returns array', () => {})
```

## 测试结构规范

### 基本结构
```typescript
describe('ComponentName', () => {
  beforeEach(() => {
    // 每个测试前的准备
  })

  afterEach(() => {
    // 每个测试后的清理
  })

  it('should do something', () => {
    // 测试逻辑
  })
})
```

### 测试准备
```typescript
beforeEach(() => {
  // 清理之前的mock
  vi.clearAllMocks()
  
  // 设置测试环境
  setActivePinia(createPinia())
  
  // 初始化组件状态
  wrapper = mount(Component, {
    global: {
      plugins: [createPinia()],
    },
  })
})
```

### 测试清理
```typescript
afterEach(() => {
  // 清理mock
  vi.restoreAllMocks()
  
  // 清理DOM
  wrapper.unmount()
})
```

## Mock使用规范

### Mock函数
```typescript
// ✅ 推荐
const mockFn = vi.fn()
const mockFn = vi.fn(() => 'default value')

// ❌ 不推荐
const mockFn = () => {}
```

### Mock模块
```typescript
// ✅ 推荐
vi.mock('@/utils/request', () => ({
  default: {
    get: vi.fn(),
    post: vi.fn(),
  },
}))

// ❌ 不推荐
const request = require('@/utils/request')
request.get = vi.fn()
```

### Mock组件
```typescript
// ✅ 推荐
vi.mock('@/components/Navbar.vue', () => ({
  name: 'Navbar',
  template: '<div class="navbar-mock">Navbar</div>'
}))

// ❌ 不推荐
const Navbar = defineComponent({
  name: 'Navbar',
  template: '<div>Navbar</div>'
})
```

## 断言使用规范

### 基本断言
```typescript
// ✅ 推荐
expect(value).toBe(expected)
expect(value).toEqual(expected)
expect(value).toContain(expected)
expect(value).toBeTruthy()
expect(value).toBeFalsy()

// ❌ 不推荐
expect(value == expected).toBe(true)
expect(value === expected).toBe(true)
```

### 对象断言
```typescript
// ✅ 推荐
expect(obj).toHaveProperty('key')
expect(obj).toMatchObject({ key: value })
expect(obj).toContainEqual({ key: value })

// ❌ 不推荐
expect(obj.key).toBe(value)
expect(obj['key']).toBe(value)
```

### 数组断言
```typescript
// ✅ 推荐
expect(array).toHaveLength(5)
expect(array).toContain(item)
expect(array).toContainEqual([item1, item2])

// ❌ 不推荐
expect(array.length).toBe(5)
expect(array.includes(item)).toBe(true)
```

## 异步测试规范

### Promise测试
```typescript
// ✅ 推荐
it('should fetch data successfully', async () => {
  const result = await fetchData()
  expect(result).toBeDefined()
})

// ❌ 不推荐
it('should fetch data successfully', () => {
  fetchData().then(result => {
    expect(result).toBeDefined()
  })
})
```

### 错误处理测试
```typescript
// ✅ 推荐
it('should handle error correctly', async () => {
  await expect(fetchData()).rejects.toThrow('Error message')
})

// ❌ 不推荐
it('should handle error correctly', async () => {
  try {
    await fetchData()
  } catch (error) {
    expect(error.message).toBe('Error message')
  }
})
```

## 组件测试规范

### 组件挂载
```typescript
// ✅ 推荐
const wrapper = mount(Component, {
  global: {
    plugins: [createPinia()],
  },
  props: {
    title: 'Test Title',
  },
})

// ❌ 不推荐
const wrapper = mount(Component)
wrapper.setProps({ title: 'Test Title' })
```

### 组件交互
```typescript
// ✅ 推荐
const button = wrapper.find('.submit-button')
await button.trigger('click')
expect(wrapper.emitted('submit')).toBeTruthy()

// ❌ 不推荐
wrapper.vm.submit()
```

### DOM查询
```typescript
// ✅ 推荐
const element = wrapper.find('.class-name')
const elements = wrapper.findAll('.class-name')

// ❌ 不推荐
const element = wrapper.vm.$el.querySelector('.class-name')
```

## 状态管理测试规范

### Store初始化
```typescript
// ✅ 推荐
it('should initialize with default state', () => {
  const store = useStore()
  expect(store.state).toEqual(defaultState)
})

// ❌ 不推荐
it('should initialize', () => {
  const store = useStore()
  expect(store).toBeDefined()
})
```

### Store操作
```typescript
// ✅ 推荐
it('should add item correctly', () => {
  const store = useStore()
  store.addItem(item)
  expect(store.items).toContainEqual(item)
})

// ❌ 不推荐
it('should add item', () => {
  const store = useStore()
  store.addItem(item)
  expect(store.items.length).toBeGreaterThan(0)
})
```

## API测试规范

### API调用测试
```typescript
// ✅ 推荐
it('should call API with correct parameters', async () => {
  const mockResponse = { data: { id: 1 } }
  vi.mocked(api).get.mockResolvedValue(mockResponse)
  
  const result = await api.getItem(1)
  expect(api.get).toHaveBeenCalledWith('/api/item/1')
  expect(result).toEqual(mockResponse.data)
})

// ❌ 不推荐
it('should call API', async () => {
  const result = await api.getItem(1)
  expect(result).toBeDefined()
})
```

### 错误处理测试
```typescript
// ✅ 推荐
it('should handle 404 error', async () => {
  const mockError = {
    response: { status: 404, data: { message: 'Not found' } }
  }
  vi.mocked(api).get.mockRejectedValue(mockError)
  
  await expect(api.getItem(999)).rejects.toThrow()
})

// ❌ 不推荐
it('should handle error', async () => {
  try {
    await api.getItem(999)
  } catch (error) {
    expect(error).toBeDefined()
  }
})
```

## 覆盖率目标

### 覆盖率标准
- **语句覆盖率（Statements）**: ≥ 70%
- **分支覆盖率（Branches）**: ≥ 70%
- **函数覆盖率（Functions）**: ≥ 70%
- **行覆盖率（Lines）**: ≥ 70%

## CI/CD集成

### GitHub Actions配置
```yaml
# .github/workflows/test.yml
name: Test

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3

          node-version: '18'
```

## 测试最佳实践总结

### DO
1. ✅ 使用描述性的测试名称
2. ✅ 每个测试只测试一个功能点
3. ✅ 使用beforeEach和afterEach进行准备和清理
4. ✅ 使用Mock隔离外部依赖
5. ✅ 测试异步操作使用async/await
6. ✅ 使用合适的断言方法
7. ✅ 测试边界条件和异常情况
8. ✅ 保持测试代码简洁和可读

### DON'T
1. ❌ 不要测试第三方库
2. ❌ 不要测试实现细节，测试行为
3. ❌ 不要在测试中包含业务逻辑
4. ❌ 不要使用硬编码的测试数据
5. ❌ 不要忽略错误处理测试
6. ❌ 不要编写过于复杂的测试
7. ❌ 不要在测试中使用随机数据
8. ❌ 不要跳过必要的测试
